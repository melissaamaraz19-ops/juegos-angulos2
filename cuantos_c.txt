  <script>
    const rondas = [
      { tipo: "rectos", correctos: 3 },
      { tipo: "agudos", correctos: 3 },
      { tipo: "obtusos", correctos: 2 },
      { tipo: "llanos", correctos: 1 }
    ];

    let rondaActual = 0;

    function lanzarConfeti() {
      confetti({
        particleCount: 150,
        spread: 70,
        origin: { y: 0.6 }
      });
    }

    function mostrarRonda() {
      const ronda = rondas[rondaActual];
      document.getElementById("pregunta").textContent = `√Ångulos ${ronda.tipo}:`;
      document.getElementById("respuesta").value = "";
      document.getElementById("result").textContent = "";
    }

    function verificar() {
      const entrada = parseInt(document.getElementById("respuesta").value);
      const ronda = rondas[rondaActual];

      if (entrada === ronda.correctos) {
        document.getElementById("result").textContent = "‚úÖ ¬°Correcto!";
        lanzarConfeti();
      } else {
        document.getElementById("result").textContent = `Incorrecto. Hay ${ronda.correctos} √°ngulos ${ronda.tipo}.`;
      }

      rondaActual++;
      if (rondaActual < rondas.length) {
        setTimeout(mostrarRonda, 2000);
      } else {
        setTimeout(() => {
          document.getElementById("pregunta").textContent = "üéâ ¬°Juego terminado!";
          document.getElementById("respuesta").style.display = "none";
          document.querySelector("button").style.display = "none";
          document.getElementById("reiniciar").style.display = "inline-block";
        }, 2500);
      }
    }

    function reiniciarJuego() {
      rondaActual = 0;
      document.getElementById("respuesta").style.display = "inline-block";
      document.querySelector("button").style.display = "inline-block";
      document.getElementById("reiniciar").style.display = "none";
      mostrarRonda();
    }

    mostrarRonda();
  </script>
